We want to ignore everything till we find "#Oracle"

What it could look like


class JSON extends JavaTokenParsers {
  def value : Parser[Any] = obj | arr | stringLiteral |
        floatingPointNumber |
        "null" | "true" | "false"
  def obj : Parser[Any] = "{"~repsep(member, ",")~"}"
  def arr : Parser[Any] = "["~repsep(value, ",")~"]"
  def member: Parser[Any] = stringLiteral~":"~value ^^ {
    case (sl, v) => (oracleFilter(sl,v))
  }

  //oracle filter
  def oracleFilter(sl: String, v: String) =
    if(sl == "text" && v.contains("Oracle")) parseObj else ignore
}
